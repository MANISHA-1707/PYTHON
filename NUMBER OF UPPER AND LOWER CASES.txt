# Python function to calculate number of upper and lower cases.

def upperlower(string):
   upper = sum(1 for char in string if char.isupper())
   lower = sum(1 for char in string if char.islower())
   print('Lower case characters = ',lower)
   print('Upper case characters = ',upper)
string = str(input("Enter a string: "))
upperlower(string)
